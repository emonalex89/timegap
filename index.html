শুরু করতে নিচের লেভেলের "Complete" চাপুন
00:00
// স্ক্রিপ্ট কন্টিনিউয়েশন... const totalLevels = 150; const levelsBox = document.getElementById('levelsBox'); const instruction = document.getElementById('instruction'); const countdownDisplay = document.getElementById('countdown'); const alarm = document.getElementById('alarmSound'); let timerInterval; // লেভেল লিস্ট জেনারেট করা for (let i = 1; i <= totalLevels; i++) { let row = document.createElement('div'); row.className = 'level-row'; row.id = `row-${i}`; row.innerHTML = ` লেভেল ${i} (সেফ টাইম: ${Math.round(getSafeTimeInSeconds(i)/60)} মি.) `; levelsBox.appendChild(row); } function startTimer(level) { // সব বাটন লক করা যেন একসাথে ২টা লেভেল ক্লিক না হয় toggleButtons(true); let timeLeft = getSafeTimeInSeconds(level); instruction.innerText = `লেভেল ${level} হ্যাক কাউন্ট হচ্ছে। অপেক্ষা করুন...`; instruction.style.color = "#ffeb3b"; timerInterval = setInterval(() => { let minutes = Math.floor(timeLeft / 60); let seconds = timeLeft % 60; countdownDisplay.innerText = `${minutes.toString().padStart(2, '0')}:${seconds.toString().padStart(2, '0')}`; if (timeLeft <= 0) { clearInterval(timerInterval); playAlarm(); // বর্তমান লেভেল কমপ্লিট মার্ক করা document.getElementById(`row-${level}`).classList.add('done'); // পরবর্তী লেভেলের নির্দেশ let nextLevel = level + 1; instruction.innerText = `বুলবুল ভাই, এবার লেভেল ${nextLevel} কমপ্লিট করো!`; instruction.style.color = "#4CAF50"; countdownDisplay.innerText = "READY!"; // বিশেষ ব্রেক চেক checkSpecialBreak(level, nextLevel); } timeLeft--; }, 1000); } function toggleButtons(status) { const buttons = document.querySelectorAll('.level-container button'); buttons.forEach(btn => btn.disabled = status); } function playAlarm() { alarm.play().catch(e => console.log("Audio play error: ", e)); setTimeout(() => { alarm.pause(); alarm.currentTime = 0; }, 4000); // ৪ সেকেন্ড পর অ্যালার্ম বন্ধ } function checkSpecialBreak(currentLevel, nextLevel) { let breakTime = 0; let msg = ""; if (currentLevel === 20) { breakTime = 15; msg = "২০ লেভেল শেষ! ১৫ মিনিটের ব্রেক দিন।"; } else if (currentLevel === 50) { breakTime = 30; msg = "৫০ লেভেল শেষ! ৩০ মিনিটের বড় ব্রেক দিন।"; } else if (currentLevel === 80) { breakTime = 60; msg = "৮০ লেভেল শেষ! ১ ঘণ্টার মেইন ব্রেক দিন।"; } else if (currentLevel === 120) { breakTime = 120; msg = "১২০ লেভেল শেষ! ২ ঘণ্টার লম্বা ব্রেক দিন।"; } if (breakTime > 0) { toggleButtons(true); // বাটন লকড থাকবে instruction.innerText = `⚠️ ${msg}`; instruction.style.color = "#ff1744"; let breakSeconds = breakTime * 60; let breakInterval = setInterval(() => { let m = Math.floor(breakSeconds / 60); let s = breakSeconds % 60; countdownDisplay.innerText = `ব্রেক: ${m}:${s}`; if (breakSeconds <= 0) { clearInterval(breakInterval); toggleButtons(false); instruction.innerText = `বুলবুল ভাই, এবার লেভেল ${nextLevel} কমপ্লিট করো!`; instruction.style.color = "#4CAF50"; countdownDisplay.innerText = "READY!"; } breakSeconds--; }, 1000); } else { toggleButtons(false); // সাধারণ লেভেলে বাটন আনলক হবে } }
